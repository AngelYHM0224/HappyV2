Depedencias 
//Ejecuta el siguiente comando para instalar todas las dependencias listadas en el package.json
npm install
// Para instalar iconos
npm install @expo/vector-icons@^14.0.4

// Para almacenamiento local
npm install @react-native-async-storage/async-storage@^1.24.0

// Para vistas enmascaradas (Masked View)
npm install @react-native-community/masked-view@^0.1.11

// Para la barra de navegación inferior
npm install @react-navigation/bottom-tabs@^6.6.1

// Para la navegación básica
npm install @react-navigation/native@^6.1.18

// Para la navegación tipo stack
npm install @react-navigation/stack@^6.4.1

// Para hacer peticiones HTTP
npm install axios@^1.7.7

// Para manejar variables de entorno
npm install dotenv@^16.4.5

// Instalación de Expo
npm install expo@~51.0.28

// Para la barra de estado de Expo
npm install expo-status-bar@~1.12.1

// Para la integración con Firebase
npm install firebase@^10.14.1

// Para la integración con la API de OpenAI
npm install openai@^4.67.3

// React para el desarrollo de interfaces
npm install react@18.2.0

// React Native para aplicaciones móviles
npm install react-native@0.74.5

// Para manejar gestos en la app
npm install react-native-gesture-handler@~2.16.1

// Para crear animaciones fluidas
npm install react-native-reanimated@~3.10.1

// Para manejar áreas seguras de la pantalla
npm install react-native-safe-area-context@4.10.5

// Para optimizar transiciones de pantalla
npm install react-native-screens@3.31.1
